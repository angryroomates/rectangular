angular.module('userApp',[]).controller('userController',function($scope){ $scope.users=[ {name:"abcd", email:"abcd@gmail.com"}, {name: "traine", email: "traine@yahoo.com"}]; $scope.newUser = {}; $scope.createUser = function(){ $scope.users.push($scope.newUser); $scope.newUser={}; }; $scope.editUser = function(user){ $scope.editedUser = user; }; $scope.updatedUser = function(){ $scope.editedUser={}; } $scope.deleteUser = function(user){ $scope.users.splice($scope.users.indexOf(user),1); }; });
